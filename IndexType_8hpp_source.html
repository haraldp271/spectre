109 namespace Tensor_detail {
110 /// \ingroup Tensor
111 /// A ::TensorIndexType holds information about what type of index an index of a
112 /// Tensor is. It holds the information about the number of spatial dimensions,
113 /// whether the index is covariant or contravariant (::UpLo), the ::Frame the
114 /// index is in, and whether the Index is Spatial or Spacetime.
115 /// \tparam SpatialDim the spatial dimensionality of the TensorIndex
116 /// \tparam Ul either UpLo::Up or UpLo::Lo for contra or covariant
117 /// \tparam Fr the Frame the TensorIndex is in
118 /// \tparam Index either IndexType::Spatial or IndexType::Spacetime
119 template <size_t SpatialDim, UpLo Ul, typename Fr, IndexType Index>
120 struct TensorIndexType {
121  static_assert(SpatialDim > 0,
122  "Cannot have a spatial dimensionality less than 1 (one) in a "
123  "TensorIndexType");
124  using value_type = decltype(SpatialDim);
125  static constexpr value_type dim =
126  Index == IndexType::Spatial ? SpatialDim
127  : SpatialDim + static_cast<value_type>(1);
128  // value is here just so that some generic metafunctions can retrieve the dim
129  // easily
130  static constexpr value_type value = dim;
131  static constexpr UpLo ul = Ul;
132  using Frame = Fr;
133  static constexpr IndexType index_type = Index;
134 };
135 } // namespace Tensor_detail
