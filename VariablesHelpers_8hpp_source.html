20 /*!
21  * \ingroup DataStructures
22  * \brief Slices the data within `vars` to a codimension 1 slice. The
23  * slice has a constant logical coordinate in direction `sliced_dim`,
24  * slicing the volume at `fixed_index` in that dimension. For
25  * example, to get the lower boundary of `sliced_dim`, pass `0` for
26  * `fixed_index`; to get the upper boundary, pass
27  * `extents[sliced_dim] - 1`.
28  *
29  * \see add_slice_to_data
30  *
31  * \return Variables class sliced to a hypersurface.
32  */
33 template <std::size_t VolumeDim, typename TagsList>
34 Variables<TagsList> data_on_slice(const Variables<TagsList>& vars,
35  const Index<VolumeDim>& element_extents,
36  const size_t sliced_dim,
37  const size_t fixed_index) noexcept {
38  const size_t interface_grid_points =
39  element_extents.slice_away(sliced_dim).product();
40  const size_t volume_grid_points = vars.number_of_grid_points();
41  constexpr const size_t number_of_independent_components =
42  Variables<TagsList>::number_of_independent_components;
43  Variables<TagsList> interface_vars(interface_grid_points);
44  const double* vars_data = vars.data();
45  double* interface_vars_data = interface_vars.data();
46  for (SliceIterator si(element_extents, sliced_dim, fixed_index); si; ++si) {
47  for (size_t i = 0; i < number_of_independent_components; ++i) {
48  // clang-tidy: do not use pointer arithmetic
49  interface_vars_data[si.slice_offset() + // NOLINT
50  i * interface_grid_points] = // NOLINT
51  vars_data[si.volume_offset() + i * volume_grid_points]; // NOLINT
52  }
53  }
54  return interface_vars;
55 }

57 /*!
58  * \ingroup DataStructures
59  * \brief Adds data on a codimension 1 slice to a volume quantity. The
60  * slice has a constant logical coordinate in direction `sliced_dim`,
61  * slicing the volume at `fixed_index` in that dimension. For
62  * example, to add to the lower boundary of `sliced_dim`, pass `0` for
63  * `fixed_index`; to add to the upper boundary, pass
64  * `extents[sliced_dim] - 1`.
65  *
66  * \see data_on_slice
67  */
68 template <std::size_t VolumeDim, typename TagsList>
69 void add_slice_to_data(const gsl::not_null<Variables<TagsList>*> volume_vars,
70  const Variables<TagsList>& vars_on_slice,
71  const Index<VolumeDim>& extents,
72  const size_t sliced_dim,
73  const size_t fixed_index) noexcept {
74  constexpr const size_t number_of_independent_components =
75  Variables<TagsList>::number_of_independent_components;
76  const size_t volume_grid_points = extents.product();
77  const size_t slice_grid_points = extents.slice_away(sliced_dim).product();
78  double* const volume_data = volume_vars->data();
79  const double* const slice_data = vars_on_slice.data();
80  for (SliceIterator si(extents, sliced_dim, fixed_index); si; ++si) {
81  for (size_t i = 0; i < number_of_independent_components; ++i) {
82  // clang-tidy: do not use pointer arithmetic
83  volume_data[si.volume_offset() + i * volume_grid_points] += // NOLINT
84  slice_data[si.slice_offset() + i * slice_grid_points]; // NOLINT
85  }
86  }
87 }

107 /// \ingroup DataStructures
108 /// Orients variables on a slice to the data-storage order of a neighbor with
109 /// the given orientation.
110 template <size_t VolumeDim, typename TagsList>
111 Variables<TagsList> orient_variables_on_slice(
112  const Variables<TagsList>& variables_on_slice,
113  const Index<VolumeDim - 1>& slice_extents, const size_t sliced_dim,
114  const Orientation<VolumeDim>& orientation_of_neighbor) noexcept {
115  const size_t number_of_grid_points = slice_extents.product();
116 
117  Variables<TagsList> oriented_variables(number_of_grid_points);
118  const auto oriented_offset = OrientVariablesOnSlice_detail::oriented_offset(
119  slice_extents, sliced_dim, orientation_of_neighbor);
120 
121  tmpl::for_each<TagsList>([&oriented_variables, &variables_on_slice,
122  &oriented_offset,
123  &number_of_grid_points](auto tag) {
124  using Tag = tmpl::type_from<decltype(tag)>;
125  auto& oriented_tensor = get<Tag>(oriented_variables);
126  const auto& tensor_on_slice = get<Tag>(variables_on_slice);
127  for (decltype(auto) oriented_and_slice_tensor_components :
128  boost::combine(oriented_tensor, tensor_on_slice)) {
129  DataVector& oriented_tensor_component =
130  boost::get<0>(oriented_and_slice_tensor_components);
131  const DataVector& tensor_component_on_slice =
132  boost::get<1>(oriented_and_slice_tensor_components);
133  for (size_t s = 0; s < number_of_grid_points; ++s) {
134  oriented_tensor_component[oriented_offset[s]] =
135  tensor_component_on_slice[s];
136  }
137  }
138  });
139 
140  return oriented_variables;
141 }
