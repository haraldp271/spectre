559 template <typename VT, bool VF>
560 DataVector& DataVector::operator=(const blaze::Vector<VT, VF>& expression) {
561  if (owning_ and (~expression).size() != size()) {
562  size_ = (~expression).size();
563  owned_data_ = InternalStorage_t(size_);
564  data_ = decltype(data_){owned_data_.data(), size_};

566  ASSERT((~expression).size() == size(), "Must copy into same size");
567  }
