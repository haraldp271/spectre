34 /// \ingroup NumericalAlgorithms
35 /// \brief Compute the partial derivatives of each variable with respect to
36 /// the coordinates of `DerivativeFrame`.
37 template <typename DerivativeTags, typename VariableTags, size_t Dim,
38  typename DerivativeFrame>
39 auto partial_derivatives(
40  const Variables<VariableTags>& u, const Index<Dim>& extents,
41  const Tensor<DataVector, tmpl::integral_list<std::int32_t, 2, 1>,
42  typelist<SpatialIndex<Dim, UpLo::Up, Frame::Logical>,
43  SpatialIndex<Dim, UpLo::Lo, DerivativeFrame>>>&
44  inverse_jacobian) {
45  const auto logical_partial_derivatives_of_u =
46  logical_partial_derivatives<DerivativeTags>(u, extents);
47 
48  Variables<db::wrap_tags_in<Tags::d, DerivativeTags, tmpl::size_t<Dim>,
49  DerivativeFrame>>
50  partial_derivatives_of_u(u.number_of_grid_points(), 0.0);
51 
52  tmpl::for_each<DerivativeTags>([
53  &partial_derivatives_of_u, &inverse_jacobian,
54  &logical_partial_derivatives_of_u
55  ](auto tag) noexcept {
56  using Tag = tmpl::type_from<decltype(tag)>;
57  using DerivativeTag = Tags::d<Tag, tmpl::size_t<Dim>, DerivativeFrame>;
58  auto& partial_derivatives_of_variable =
59  get<DerivativeTag>(partial_derivatives_of_u);
60  for (auto it = partial_derivatives_of_variable.begin();
61  it != partial_derivatives_of_variable.end(); ++it) {
62  const auto deriv_indices =
63  partial_derivatives_of_variable.get_tensor_index(it);
64  const size_t deriv_index = deriv_indices[0];
65  const auto tensor_indices =
66  all_but_specified_element_of<0>(deriv_indices);
67  for (size_t d = 0; d < Dim; ++d) {
68  *it += inverse_jacobian.get(d, deriv_index) *
69  get<Tag>(gsl::at(logical_partial_derivatives_of_u, d))
70  .get(tensor_indices);
71  }
72  }
73  });
74 
75  return partial_derivatives_of_u;
76 }

79 template <typename VariableTags, typename DerivativeTags>
80 struct LogicalImpl<1, VariableTags, DerivativeTags> {
81  static constexpr const size_t Dim = 1;
82  static auto apply(const Variables<VariableTags>& u,
83  const Index<Dim>& extents) {
84  auto logical_partial_derivatives_of_u = make_array<Dim>(
85  Variables<DerivativeTags>(u.number_of_grid_points(), 0.0));
86  const Matrix& differentiation_matrix_xi =
87  Basis::lgl::differentiation_matrix(extents[0]);
88  dgemm_('N', 'N', extents[0],
89  logical_partial_derivatives_of_u[0].size() / extents[0], extents[0],
90  1.0, differentiation_matrix_xi.data(), extents[0], u.data(),
91  extents[0], 0.0, logical_partial_derivatives_of_u[0].data(),
92  extents[0]);
93 
94  return logical_partial_derivatives_of_u;
95  }
96 };
