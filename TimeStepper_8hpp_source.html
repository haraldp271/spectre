61  /// Add the change for the current substep to u. New values should
62  /// be pushed onto the end of the history when evaluated and
63  /// obsolete values should be removed from the front, in a manner
64  /// similar to a queue.
65  template <typename Vars, typename DerivVars>
66  void update_u(const gsl::not_null<Vars*> u,
67  const std::deque<std::tuple<Time, Vars, DerivVars>>& history,
68  const TimeDelta& time_step) const noexcept {
69  return TimeStepper_detail::fake_virtual_update_u<creatable_classes>(
70  this, u, history, time_step);
71  }

73  /// \brief Compute the change in a boundary quantity due to the
74  /// coupling on the interface.
75  ///
76  /// The `history` is a vector representing the different sides of
77  /// the interface, with the local side first. Each entry in that
78  /// vector is the time history of a side, similar to the argument to
79  /// `update_u`.
80  ///
81  /// The coupling function `coupling` should compute the derivative
82  /// of the boundary quantity from the side data, supplied as a
83  /// `std::vector<std::reference_wrapper<const FluxVars>>`. These
84  /// values may be used to form a linear combination internally, so
85  /// `BoundaryVars` should have appropriate mathematical operators
86  /// defined to allow that.
87  template <typename BoundaryVars, typename FluxVars, typename Coupling>
88  BoundaryVars compute_boundary_delta(
89  const Coupling& coupling,
90  const std::vector<std::deque<std::tuple<Time, BoundaryVars, FluxVars>>>&
91  history,
92  const TimeDelta& time_step) const noexcept {
93  static_assert(
94  std::is_convertible<
95  std::decay_t<std::result_of_t<const Coupling&(
96  const std::vector<std::reference_wrapper<const FluxVars>>&)>>,
97  BoundaryVars>::value,
98  "Coupling function returns wrong type");
99  return TimeStepper_detail::fake_virtual_compute_boundary_delta<
100  creatable_classes>(this, coupling, history, time_step);
101  }
102 
103  /// Return iterator to the first entry in `history` that is still
104  /// needed after the current step. Entries up to that point must be
105  /// removed before the next call to `update_u`.
106  template <typename Vars, typename DerivVars>
107  typename std::deque<std::tuple<Time, Vars, DerivVars>>::const_iterator
108  needed_history(const std::deque<std::tuple<Time, Vars, DerivVars>>& history)
109  const noexcept {
110  return TimeStepper_detail::fake_virtual_needed_history<creatable_classes>(
111  this, history);
112  }
113 
114  /// Return iterators to the first entry for each element of
115  /// `history` that is still needed after the current step. Entries
116  /// up to that point must be removed before the next call to
117  /// `compute_boundary_delta`.
118  template <typename BoundaryVars, typename FluxVars>
119  typename std::vector<typename std::deque<

121  needed_boundary_history(
122  const std::vector<
123  std::deque<std::tuple<Time, BoundaryVars, FluxVars>>>& history,
124  const TimeDelta& time_step) const noexcept {
125  return TimeStepper_detail::fake_virtual_needed_boundary_history<
126  creatable_classes>(this, history, time_step);
127  }
