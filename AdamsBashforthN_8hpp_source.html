75  /*!
76  * An explanation of the computation being performed by this
77  * function:
78  *
79  * We number the sides \f$1\f$ through \f$S\f$, and let the
80  * evaluation times on side \f$s\f$ be \f$\ldots, t^s_{-1}, t^s_0,
81  * t^s_1, \ldots\f$, with the starting location of the numbering
82  * arbitrary. The union of all these sets of times produces the
83  * sequence \f$\ldots, \tilde{t}_{-1}, \tilde{t}_0, \tilde{t}_1,
84  * \ldots\f$, again numbered arbitrarily. For example, one possible
85  * sequence of times for a two-sided boundary is:
86  * \f{equation}
87  * \begin{aligned}
88  * \text{Side 1:} \\ \text{Union times:} \\ \text{Side 2:}
89  * \end{aligned}
90  * \cdots
91  * \begin{gathered}
92  * t^1_4 \\ \tilde{t}_8 \\ t^2_1
93  * \end{gathered}
94  * \leftarrow \Delta \tilde{t}_8 \rightarrow
95  * \begin{gathered}
96  * \, \\ \tilde{t}_9 \\ t^2_2
97  * \end{gathered}
98  * \leftarrow \Delta \tilde{t}_9 \rightarrow
99  * \begin{gathered}
100  * t^1_5 \\ \tilde{t}_{10} \\ \,
101  * \end{gathered}
102  * \cdots
103  * \f}
104  *
105  * We call the indices of the start and end times in the union time
106  * set \f$n_S\f$ and \f$n_E\f$, respectively, so for the above
107  * example, if we wish to compute the step on side 1 from
108  * \f$t^1_4\f$ to \f$t^1_5\f$, we would have \f$n_S = 8\f$ and
109  * \f$n_E = 10\f$. The indices of steps on each side containing or
110  * starting with union time number \f$n\f$ are denoted
111  * \f$\vec{m}_n\f$, so in the above example \f$\vec{m}_9 = (4,
112  * 2)\f$. The union indices corresponding to the start of a set of
113  * side step indices \f$\vec{m}\f$ are denoted
114  * \f$\vec{n}_{\vec{m}}\f$, so in the example \f$\vec{n}_{(4, 2)} =
115  * (8, 9)\f$. In the below equations, addition, subtraction, min,
116  * and max operations applied to vectors operate on each component
117  * individually, i.e., \f$\vec{q} + k = (q^1 + k, \ldots, q^S +
118  * k)\f$.
119  *
120  * Let \f$k\f$ be the order of the integrator. For a sequence of
121  * times \f$t^s_{j_s - (k-1)}, \ldots, t^s_{j_s}\f$ on each side
122  * \f$s\f$, we can write the cardinal functions on those times as
123  * \f{equation}{
124  * \newcommand\of[1]{\!\!\left(#1\right)}
125  * C_{a_1, \ldots, a_S}(t_1, \ldots, t_S; j_1, \ldots, j_S) =
126  * \prod_{s=1}^S \ell_{a_s}\of{t_s; t^s_{j_s - (k-1)}, \ldots, t^s_{j_s}},
127  * \f}
128  * with \f$\ell_a(t; x_1, \ldots, x_k)\f$ the Lagrange interpolating
129  * polynomials. (Note that we take the indices \f$a\f$ to run from
130  * \f$1\f$ to \f$k\f$ here, but from \f$0\f$ to \f$k-1\f$ in the
131  * code.) The cardinal function satisfies
132  * \f$ C_{a_1, \ldots, a_S}(t^1_{i_1}, \ldots, t^S_{i_S}; j_1, \ldots, j_S) =
133  * \prod_{s=1}^S \delta_{i_s, j_s + a_s - k} \f$.
134  * We write the integral of a cardinal function using an
135  * Adams-Bashforth step based on the union times along the diagonal
136  * \f$t_1 = t_2 = \cdots = t_S\f$ from \f$\tilde{t}_n\f$ to
137  * \f$\tilde{t}_{n+1}\f$ as
138  * \f{equation}{
139  * \newcommand\of[1]{\!\!\left(#1\right)}
140  * I_{n,\vec{q}} =
141  * \Delta \tilde{t}_n
142  * \sum_{j = 0}^{k-1}
143  * \tilde{\alpha}_{nj}
144  * C_{\vec{q}-\vec{m}_n+k}
145  * \of{\tilde{t}_{n - j}, \ldots, \tilde{t}_{n - j}; \vec{m}_n},
146  * \f}
147  * where \f$\tilde{\alpha}_{nj}\f$ are the Adams-Bashforth
148  * coefficients derived from the union times and \f$\vec{q}\f$ is a
149  * vector specifying the cardinal function indices. The boundary
150  * delta for the step is then
151  * \f{equation}{
152  * \newcommand\mat{\mathbf}
153  * \mat{F}_{n_S, n_E} =
154  * \mspace{-7mu}
155  * \sum_{\vec{q} = \vec{m}_{n_s}-(k-1)}^{\vec{m}_{n_E-1}}
156  * \mspace{-4mu}
157  * \mat{D}_{\vec{q}}
158  * \mspace{-4mu}
159  * \sum_{n = \max\left\{n_S, \vec{n}_{\vec{q}} \right\}}
160  * ^{\min\left\{n_E, \vec{n}_{\vec{q} + k}\right\} - 1}
161  * \mspace{-7mu}
162  * I_{n,\vec{q}},
163  * \f}
164  * where \f$\mat{D}_{\vec{q}}\f$ is the coupling function evaluated
165  * at the side values at side step indices \f$\vec{q}\f$.
166  */

264 template <typename Vars, typename DerivVars>
265 void AdamsBashforthN::update_u(
266  const gsl::not_null<Vars*> u,
267  const std::deque<std::tuple<Time, Vars, DerivVars>>& history,
268  const TimeDelta& time_step) const noexcept {
269  ASSERT(is_self_starting_ or target_order_ == history.size(),
270  "Length of history should be the order, so "
271  << target_order_ << ", but is: " << history.size());
272  ASSERT(history.size() <= target_order_,
273  "Length of history (" << history.size() << ") "
274  << "should not exceed target order (" << target_order_ << ")");
275 
276  const auto& coefficients = get_coefficients(time_iter(history.cbegin()),
277  time_iter(history.cend()),
278  time_step);
279 
280  switch (history.size()) {
281  case 1:
282  *u += time_step.value() * std::get<2>(history[0]);
283  return;
284  case 2:
285  *u += time_step.value() * (coefficients[1] * std::get<2>(history[0]) +
286  coefficients[0] * std::get<2>(history[1]));
287  return;
288  case 3:
289  *u += time_step.value() * (coefficients[2] * std::get<2>(history[0]) +
290  coefficients[1] * std::get<2>(history[1]) +
291  coefficients[0] * std::get<2>(history[2]));
292  return;
293  case 4:
294  *u += time_step.value() * (coefficients[3] * std::get<2>(history[0]) +
295  coefficients[2] * std::get<2>(history[1]) +
296  coefficients[1] * std::get<2>(history[2]) +
297  coefficients[0] * std::get<2>(history[3]));
298  return;
299  case 5:
300  *u += time_step.value() * (coefficients[4] * std::get<2>(history[0]) +
301  coefficients[3] * std::get<2>(history[1]) +
302  coefficients[2] * std::get<2>(history[2]) +
303  coefficients[1] * std::get<2>(history[3]) +
304  coefficients[0] * std::get<2>(history[4]));
305  return;
306  case 6:
307  *u += time_step.value() * (coefficients[5] * std::get<2>(history[0]) +
308  coefficients[4] * std::get<2>(history[1]) +
309  coefficients[3] * std::get<2>(history[2]) +
310  coefficients[2] * std::get<2>(history[3]) +
311  coefficients[1] * std::get<2>(history[4]) +
312  coefficients[0] * std::get<2>(history[5]));
313  return;
314  case 7:
315  *u += time_step.value() * (coefficients[6] * std::get<2>(history[0]) +
316  coefficients[5] * std::get<2>(history[1]) +
317  coefficients[4] * std::get<2>(history[2]) +
318  coefficients[3] * std::get<2>(history[3]) +
319  coefficients[2] * std::get<2>(history[4]) +
320  coefficients[1] * std::get<2>(history[5]) +
321  coefficients[0] * std::get<2>(history[6]));
322  return;
323  case 8:
324  *u += time_step.value() * (coefficients[7] * std::get<2>(history[0]) +
325  coefficients[6] * std::get<2>(history[1]) +
326  coefficients[5] * std::get<2>(history[2]) +
327  coefficients[4] * std::get<2>(history[3]) +
328  coefficients[3] * std::get<2>(history[4]) +
329  coefficients[2] * std::get<2>(history[5]) +
330  coefficients[1] * std::get<2>(history[6]) +
331  coefficients[0] * std::get<2>(history[7]));
332  return;
333  default:
334  ERROR("Bad amount of history data: " << history.size());
335  }

338 template <typename BoundaryVars, typename FluxVars, typename Coupling>
339 BoundaryVars AdamsBashforthN::compute_boundary_delta(
340  const Coupling& coupling,
341  const std::vector<std::deque<std::tuple<Time, BoundaryVars, FluxVars>>>&
342  history,
343  const TimeDelta& time_step) const noexcept {
344  ASSERT(not is_self_starting_, "Unimplemented");
345 
346  using HistoryEntry = decltype(history[0][0]);
347  using HistoryIterator = decltype(history[0].cbegin());
348 
349  // Avoid billions of casts
350  const auto target_order_s = static_cast<ssize_t>(target_order_);
351 
352  const SimulationLess simulation_less(time_step.is_positive());
353 
354  // Cast the overloaded get function to a function pointer taking a
355  // history entry to a time reference. This is needed to prevent the
356  // boost::make_transform_iterator calls from being ambiguous.
357  const Time& (*const get_time)(const HistoryEntry&) = std::get<0>;
358 
359  // Time::value is fairly slow, so cache values.
360  auto time_value =
361  make_cached_function<Time, std::map, Time::StructuralCompare>(
362  [](const Time& t) noexcept { return t.value(); });
363 
364  const size_t num_sides = history.size();
365 
366  // Evaluate the cardinal function for a grid of points at a point on
367  // the diagonal.
368  const auto grid_cardinal_function =
369  [target_order_s, &time_value, &get_time](
370  const double evaluation_time,
371  const auto& function_index,
372  const auto& grid_times) noexcept {
373  double result = 1.;
374  for (size_t side = 0; side < grid_times.size(); ++side) {
375  // Stop calculating things if we're just going to multiply
376  // them by zero.
377  if (result == 0.) { break; }
378 
379  // Makes an iterator with a map to give time as a double.
380  const auto make_lagrange_iterator =
381  [&time_value, &get_time](const HistoryIterator& it) noexcept {
382  return boost::make_transform_iterator(
383  it, [&time_value, &get_time](const HistoryEntry& h) noexcept {
384  return time_value(get_time(h));
385  });
386  };
387  result *= lagrange_polynomial(
388  make_lagrange_iterator(function_index[side]),
389  evaluation_time,
390  make_lagrange_iterator(grid_times[side] - (target_order_s - 1)),
391  make_lagrange_iterator(grid_times[side] + 1));
392  }
393  return result;
394  };
395 
396  for (const auto& side_hist : history) {
397  ASSERT(not side_hist.empty(), "No data");
398  ASSERT(std::is_sorted(time_iter(side_hist.begin()),
399  time_iter(side_hist.end()),
400  simulation_less),
401  "History not in order");
402  }
403 
404  // Start and end of the step we are trying to take
405  const Time start_time = get_time(history[0].back());
406  const Time end_time = start_time + time_step;
407 
408  // Union of times of all step boundaries on any side.
409  const auto union_times =
410  [&end_time, &get_time, &history, &simulation_less]() noexcept {
411  std::set<Time, decltype(simulation_less)> ret({end_time}, simulation_less);
412  for (const auto& side_hist : history) {
413  ret.insert(time_iter(side_hist.cbegin()), time_iter(side_hist.cend()));
414  ASSERT(simulation_less(get_time(side_hist.back()), end_time),
415  "Please supply only older data: " << get_time(side_hist.back())
416  << " is not before " << end_time);
417  }
418  return ret;
419  }();
420 
421  // Calculating the Adams-Bashforth coefficients is somewhat
422  // expensive, so we cache them. ab_coefs(it) returns the
423  // coefficients used to step from *it to *(it + 1).
424  auto ab_coefs = [target_order_s]() noexcept {
425  using Iter = decltype(union_times.cbegin());
426  auto compare = [](const Iter& a, const Iter& b) noexcept {
427  return Time::StructuralCompare{}(*a, *b);
428  };
429  return make_cached_function<Iter, std::map, decltype(compare)>(
430  [target_order = target_order_s](const Iter& last_time) noexcept {
431  const auto times_end = std::next(last_time);
432  const auto times_begin = std::prev(last_time, target_order - 1);
433  return get_coefficients(times_begin, times_end,
434  *times_end - *last_time);
435  },
436  std::move(compare));
437  }();
438 
439  // Result variable.
440  auto accumulated_change =
441  make_with_value<BoundaryVars>(std::get<1>(history[0].back()), 0.);
442 
443  // Ranges of values where we evaluate the coupling function.
444  std::vector<std::pair<HistoryIterator, HistoryIterator>>
445  coupling_evaluation_ranges;
446  for (size_t side = 0; side < num_sides; ++side) {
447  const auto& side_hist = history[side];
448  const auto current_step_on_side =
449  step_containing_time(side_hist, start_time, simulation_less);
450  ASSERT(std::distance(side_hist.begin(), current_step_on_side) >=
451  target_order_s - 1,
452  "Not enough data on side " << side << ": "
453  << std::distance(side_hist.begin(), current_step_on_side)
454  << " (" << get_time(side_hist.front()) << " to "
455  << get_time(*current_step_on_side) << ")");
456  coupling_evaluation_ranges.emplace_back(
457  current_step_on_side - (target_order_s - 1), side_hist.end());
458  }
459  for (auto coupling_evaluation =
460  make_general_index_iterator(std::move(coupling_evaluation_ranges));
461  coupling_evaluation;
462  ++coupling_evaluation) {
463  double deriv_coef = 0.;
464 
465  // Prepare to loop over the union times relevant for the current
466  // coupling evaluation.
467 
468  // Latest of start of current step (otherwise was computed in a
469  // previous call) and the times of the side data being coupled
470  // (since segments can't depend on later data).
471  Time coupling_start_time = start_time;
472  // Earliest of the end of the current step (since the step can't
473  // depend on later data), and target_order steps after the
474  // evaluation on each side (since AB only uses data for that
475  // long).
476  Time coupling_end_time = end_time;
477  for (size_t side = 0; side < num_sides; ++side) {
478  const auto& evaluation_side = coupling_evaluation[side];
479  if (simulation_less(coupling_start_time, get_time(*evaluation_side))) {
480  coupling_start_time = get_time(*evaluation_side);
481  }
482  if (history[side].end() - evaluation_side > target_order_s and
483  simulation_less(get_time(*(evaluation_side + target_order_s)),
484  coupling_end_time)) {
485  coupling_end_time = get_time(*(evaluation_side + target_order_s));
486  }
487  }
488  const auto union_times_for_coupling_begin =
489  std::lower_bound(union_times.cbegin(), union_times.cend(),
490  coupling_start_time, simulation_less);
491  const auto union_times_for_coupling_end =
492  std::lower_bound(union_times_for_coupling_begin, union_times.cend(),
493  coupling_end_time, simulation_less);
494 
495  // Iterators to the time on each side containing the union time we are
496  // currently considering.
497  std::vector<HistoryIterator> side_indices;
498  for (const auto& side_hist : history) {
499  side_indices.push_back(step_containing_time(
500  side_hist, *union_times_for_coupling_begin, simulation_less));
501  }
502 
503  // This loop computes the coefficient of the coupling evaluation,
504  // i.e., the sum over the I_{n,q}, as deriv_coef.
505  for (auto union_time = union_times_for_coupling_begin;
506  union_time != union_times_for_coupling_end;
507  ++union_time) {
508  // Advance the step of any side where we've moved into the next
509  // step.
510  for (size_t side = 0; side < num_sides; ++side) {
511  const auto next_index = side_indices[side] + 1;
512  if (next_index != history[side].end() and
513  not simulation_less(*union_time, get_time(*next_index))) {
514  side_indices[side] = next_index;
515  }
516  }
517 
518  // Perform a single step Adams-Bashforth integration of a grid
519  // cardinal function. This computes the quantity called I_{n,q}
520  // in the documentation as integrated_cardinal_function.
521  const std::vector<double>& method_coefs = ab_coefs(union_time);
522  // This produces a reversed iterator pointing to the same entry
523  // as union_time.
524  auto recent_time = std::make_reverse_iterator(std::next(union_time));
525  auto method_it = method_coefs.begin();
526  double integrated_cardinal_function = 0.;
527  for (; method_it != method_coefs.end(); ++method_it, ++recent_time) {
528  integrated_cardinal_function +=
529  *method_it * grid_cardinal_function(time_value(*recent_time),
530  coupling_evaluation,
531  side_indices);
532  }
533  integrated_cardinal_function *=
534  (*std::next(union_time) - *union_time).value();
535 
536  deriv_coef += integrated_cardinal_function;
537  } // for union_time
538 
539  if (deriv_coef != 0.) {
540  // We sometimes get exact zeros from the Lagrange polynomials.
541  // Skip the (potentially expensive) coupling calculation in that
542  // case.
543 
544  std::vector<std::reference_wrapper<const FluxVars>> args;
545  for (size_t side = 0; side < num_sides; ++side) {
546  args.push_back(std::cref(std::get<2>(*coupling_evaluation[side])));
547  }
548 
549  accumulated_change += deriv_coef * coupling(args);
550  }
551  } // for coupling_evaluation
552 
553  return accumulated_change;
554 }

568 template <typename BoundaryVars, typename FluxVars>
569 typename std::vector<typename std::deque<
570  std::tuple<Time, BoundaryVars, FluxVars>>::const_iterator>
571 AdamsBashforthN::needed_boundary_history(
572  const std::vector<
573  std::deque<std::tuple<Time, BoundaryVars, FluxVars>>>& history,
574  const TimeDelta& time_step) const noexcept {
575  const SimulationLess simulation_less(time_step.is_positive());
576  const Time next_start_time = std::get<0>(history[0].back()) + time_step;
577  std::vector<typename std::deque<
578  std::tuple<Time, BoundaryVars, FluxVars>>::const_iterator> result;
579  result.reserve(history.size());
580  // We know that the local side will step at next_start_time, so the
581  // step containing that time will be the next step, which is not
582  // currently in the history.
583  result.push_back(history[0].end() - static_cast<ssize_t>(target_order_ - 1));
584  // We don't know whether other sides will step at next_start_time,
585  // so we have to be conservative and assume they will not.
586  for (size_t side = 1; side < history.size(); ++side) {
587  result.push_back(step_containing_time(history[side], next_start_time,
588  simulation_less) -
589  static_cast<ssize_t>(target_order_ - 1));
590  }
591  return result;
592 }

594 template <typename Iterator>
595 std::vector<double> AdamsBashforthN::get_coefficients(
596  const Iterator& times_begin, const Iterator& times_end,
597  const TimeDelta& step) noexcept {
598  std::vector<double> steps;
599  // This may be slightly more space than we need, but we can't get
600  // the exact amount without iterating through the iterators, which
601  // is not necessarily cheap depending on the iterator type.
602  steps.reserve(maximum_order);
603  for (auto t = times_begin; std::next(t) != times_end; ++t) {
604  steps.push_back((*std::next(t) - *t) / step);
605  }
606  steps.push_back(1.);
607  return get_coefficients_impl(steps);
608 }
