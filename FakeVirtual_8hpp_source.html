14 /// \ingroup Utilities
15 /// \brief Define a function that acts similarly to a virtual
16 /// function, but can take template parameters.
17 ///
18 /// \details `DEFINE_FAKE_VIRTUAL(func)` defines the function
19 /// `fake_virtual_func` and the struct `FakeVirtualInherit_func`. It
20 /// should usually be called in a detail namespace.
21 ///
22 /// A base class `Base` using this functionality should define a type
23 /// \code
24 /// using Inherit = FakeVirtualInherit_func<Base>;
25 /// \endcode
26 /// and a member function `func` wrapping `fake_virtual_func`, with
27 /// the wrapper passing the derived classes as a typelist as the first
28 /// template argument and the `this` pointer as the first normal
29 /// argument.
30 ///
31 /// Derived classes should then inherit from `Base::Inherit` instead
32 /// of directly from `Base`. (`Base::Inherit` inherits from `Base`.)
33 ///
34 /// If the base class has no pure virtual functions remaining it will
35 /// generally be desirable to mark the constructors and assignment
36 /// operators protected so that a bare base class cannot be instantiated.
37 ///
38 /// If it is necessary to use multiple fake virtual functions with the
39 /// same base class, the `Inherit` definition can nest the fake
40 /// virtual classes:
41 /// \code
42 /// using Inherit = FakeVirtualInherit_func1<FakeVirtualInherit_func2<Base>>;
43 /// \endcode
44 ///
45 /// \example
46 /// \snippet Test_FakeVirtual.cpp fake_virtual_example
47 #define DEFINE_FAKE_VIRTUAL(function) \
48  /* This struct is only needed for producing an error if the function */ \
49  /* is not overridden in the derived class. */ \
50  template <typename Base> \
51  struct FakeVirtualInherit_##function : public Base { \
52  using Base::Base; \
53  /* clang-tidy: I think "= delete" was overlooked in the guideline */ \
54  void function(...) const = delete; /* NOLINT */ \
55  }; \
56  \
57  template <typename Classes, typename... TArgs, typename Base, \
58  typename... Args, \
59  Requires<(tmpl::size<Classes>::value == 0)> = nullptr> \
60  [[noreturn]] auto fake_virtual_##function(Base* obj, Args&&... args) \
61  /* clang-tidy: macro arg in parentheses */ \
62  ->decltype(obj->template function<TArgs...>(args...)) { /* NOLINT */\
63  ERROR("Class " << pretty_type::get_runtime_type_name(*obj) \
64  << " is not registered with " \
65  << pretty_type::get_name<std::remove_const_t<Base>>()); \
66  } \
67  \
68  template <typename Classes, typename... TArgs, typename Base, \
69  typename... Args, \
70  Requires<(tmpl::size<Classes>::value != 0)> = nullptr> \
71  decltype(auto) fake_virtual_##function(Base* obj, Args&&... args) { \
72  using derived = tmpl::front<Classes>; \
73  static_assert(std::is_base_of<typename Base::Inherit, derived>::value, \
74  "Derived class does not inherit from Base::Inherit"); \
75  using derived_p = std::conditional_t<std::is_const<Base>::value, \
76  derived const*, derived*>; \
77  /* If we want to allow creatable classses to return objects of */ \
78  /* types derived from themselves then this will have to be changed */ \
79  /* to a dynamic_cast, but we probably won't want that and this */ \
80  /* form is significantly faster. */ \
81  return typeid(*obj) == typeid(derived) \
82  ? static_cast<derived_p>(obj) \
83  /* clang-tidy: macro arg in parentheses */ \
84  ->template function<TArgs...>(/* NOLINT */ \
85  std::forward<Args>(args)...) \
86  : fake_virtual_##function<tmpl::pop_front<Classes>, TArgs...>( \
87  obj, std::forward<Args>(args)...); \
88  }
