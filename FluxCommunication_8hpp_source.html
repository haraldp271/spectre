39 /// \ingroup DiscontinuousGalerkin
40 /// \brief Receive boundary data from neighbors and compute boundary
41 /// contribution to the time derivative.
42 ///
43 /// \note The data must be oriented by the sender, but this class
44 /// handles projection, etc.
45 ///
46 /// Uses:
47 /// - ConstGlobalCache: Metavariables::numerical_flux
48 /// - DataBox: Tags::Element<volume_dim>,
49 /// Tags::HistoryBoundaryVariables<Direction<volume_dim>,
50 /// system::variables_tag>, Tags::Extents<volume_dim>, Tags::TimeId,
51 /// Tags::UnnormalizedGridNormal<volume_dim>
52 ///
53 /// DataBox changes:
54 /// - Adds: system::dt_variables_tag
55 /// - Removes: Tags::HistoryBoundaryVariables<Direction<volume_dim>,
56 /// system::variables_tag>
57 /// - Modifies: nothing
58 template <typename System>
59 struct ComputeBoundaryFlux {
60  struct FluxesTag {
61  using temporal_id = TimeId;
62  using type =
63  std::unordered_map<
64  TimeId,
65  std::unordered_map<
66  std::pair<Direction<System::volume_dim>,
67  ElementId<System::volume_dim>>,
68  db::item_type<typename System::variables_tag>,
69  boost::hash<std::pair<Direction<System::volume_dim>,

82  template <typename DbTags, typename... InboxTags, typename Metavariables,
83  typename ArrayIndex, typename ActionList>
84  static auto apply(db::DataBox<DbTags>& box,
85  tuples::TaggedTuple<InboxTags...>& inboxes,
86  const Parallel::ConstGlobalCache<Metavariables>& cache,
87  const ArrayIndex& /*array_index*/,
88  const ActionList /*meta*/) noexcept {
89  static_assert(cpp17::is_same_v<System, typename Metavariables::system>,
90  "Inconsistent systems");
91  constexpr const size_t volume_dim = System::volume_dim;
92  using variables_tag = typename System::variables_tag;
93  using dt_variables_tag = typename System::dt_variables_tag;
94 
95  const auto& flux_computer =
96  get<typename Metavariables::numerical_flux>(cache);
97  const auto& extents = db::get<Tags::Extents<volume_dim>>(box);
98  const auto& local_data =
99  db::get<Tags::HistoryBoundaryVariables<
100  Direction<volume_dim>, variables_tag>>(box);
101 
102  auto dt_vars = db::item_type<dt_variables_tag>(extents.product(), 0.);
103 
104  const auto& element = db::get<Tags::Element<volume_dim>>(box);
105 
106  auto& inbox = tuples::get<FluxesTag>(inboxes);
107  const auto& time_id = db::get<Tags::TimeId>(box);
108  auto remote_data = std::move(inbox[time_id]);
109  inbox.erase(time_id);
110 
111  for (const auto& direction_neighbors : element.neighbors()) {
112  const auto& direction = direction_neighbors.first;
113  const size_t dimension = direction.dimension();
114  const auto& neighbors_in_direction = direction_neighbors.second;
115  ASSERT(neighbors_in_direction.size() == 1,
116  "Complex mortars unimplemented");
117  const auto& neighbor = *neighbors_in_direction.begin();
118  const auto neighbor_data_it =
119  remote_data.find(std::make_pair(direction, neighbor));
120  ASSERT(remote_data.end() != neighbor_data_it,
121  "Attempting to access neighbor flux data that does not exist. "
122  "The neighbor attempting to be accessed is: "
123  << neighbor << " in direction " << direction
124  << ". The known keys are " << keys_of(remote_data) << ".");
125  const auto& local_value = local_data.at(direction);
126  const auto& neighbor_value = neighbor_data_it->second;
127 
128  // This needs to be fixed for hp-adaptivity to handle
129  // projections correctly
130 
131  const auto& grid_normal =
132  db::get<Tags::UnnormalizedGridNormal<volume_dim>>(box).at(direction);
133  // Using this instead of auto prevents incomprehensible errors
134  // if the return type of compute_flux is wrong.
135  using FluxType = db::item_type<db::add_tag_prefix<
136  Tags::Flux, variables_tag, tmpl::size_t<volume_dim>, Frame::Grid>>;
137  const FluxType local_boundary_flux =
138  System::compute_flux::apply(local_value);
139  const FluxType neighbor_boundary_flux =
140  System::compute_flux::apply(neighbor_value);
141  using normal_flux_tag =
142  db::add_tag_prefix<Tags::NormalDotFlux, variables_tag>;
143  db::item_type<normal_flux_tag> local_normal_flux(
144  local_boundary_flux.number_of_grid_points(), 0.);
145  db::item_type<normal_flux_tag> neighbor_normal_flux(
146  local_boundary_flux.number_of_grid_points(), 0.);
147  tmpl::for_each<typename variables_tag::tags_list>([
148  &local_normal_flux, &neighbor_normal_flux, &local_boundary_flux,
149  &neighbor_boundary_flux, &grid_normal
150  ](auto tag) noexcept {
151  using Tag = tmpl::type_from<decltype(tag)>;
152  using flux_tag = Tags::Flux<Tag, tmpl::size_t<2>, Frame::Grid>;
153 
154  const auto& local_bf = get<flux_tag>(local_boundary_flux);
155  auto& local_nf = get<Tags::NormalDotFlux<Tag>>(local_normal_flux);
156  const auto& neighbor_bf = get<flux_tag>(neighbor_boundary_flux);
157  auto& neighbor_nf = get<Tags::NormalDotFlux<Tag>>(neighbor_normal_flux);
158  for (auto flux_it = local_bf.begin(); flux_it != local_bf.end();
159  ++flux_it) {
160  const auto flux_index =
161  local_bf.get_tensor_index(flux_it);
162  const size_t contract_index = flux_index[0];
163  const auto other_indices =
164  all_but_specified_element_of<0>(flux_index);
165  local_nf.get(other_indices) +=
166  grid_normal.get(contract_index) * local_bf.get(flux_index);
167  neighbor_nf.get(other_indices) +=
168  grid_normal.get(contract_index) * neighbor_bf.get(flux_index);
169  }
170  });
171 
172  const tuples::TaggedTuple<variables_tag, normal_flux_tag> self_data(
173  local_value, std::move(local_normal_flux));
174  const tuples::TaggedTuple<variables_tag, normal_flux_tag> neighbor_data(
175  neighbor_value, std::move(neighbor_normal_flux));
176 
177  auto numerical_flux = apply_flux(
178  flux_computer, self_data, neighbor_data,
179  typename std::decay_t<decltype(flux_computer)>::argument_tags{});
180 
181  // Needs fixing for GH/curved
182  DataVector magnitude_of_grid_normal = magnitude(grid_normal);
183  const auto lifted_data = dg::lift_flux(
184  tuples::get<normal_flux_tag>(self_data),
185  std::move(numerical_flux),
186  extents[dimension],
187  std::move(magnitude_of_grid_normal));
188 
189  // Convert tags. This will be replaced by expanded Variables
190  // functionality. See #319.
191  std::decay_t<decltype(dt_vars)> retagged_lifted_data = 1. * lifted_data;
192 
193  add_slice_to_data(
194  make_not_null(&dt_vars), retagged_lifted_data, extents, dimension,
195  direction.side() == Side::Lower ? 0 : extents[dimension] - 1);
196  }
197 
198  return std::make_tuple(db::create_from<
199  db::RemoveTags<Tags::HistoryBoundaryVariables<
200  Direction<volume_dim>, variables_tag>>,
201  db::AddTags<dt_variables_tag>>(
202  box, std::move(dt_vars)));
203  }

205  template <typename DbTags, typename... InboxTags, typename Metavariables,
206  typename ArrayIndex>
207  static bool is_ready(
208  const db::DataBox<DbTags>& box,
209  const tuples::TaggedTuple<InboxTags...>& inboxes,
210  const Parallel::ConstGlobalCache<Metavariables>& /*cache*/,
211  const ArrayIndex& /*array_index*/) noexcept {
212  const auto& time_id = db::get<Tags::TimeId>(box);
213  const auto& inbox = tuples::get<FluxesTag>(inboxes);
214  auto receives = inbox.find(time_id);
215  const auto number_of_neighbors =
216  db::get<Tags::Element<Metavariables::system::volume_dim>>(box)
217  .number_of_neighbors();
218  const size_t num_receives =
219  receives == inbox.end() ? 0 : receives->second.size();
220  return num_receives == number_of_neighbors;
221  }

224 /// \ingroup ActionsGroup
225 /// \ingroup DiscontinuousGalerkin
226 /// \brief Compute local boundary data needed for fluxes and send it
227 /// to neighbors.
228 ///
229 /// \note The data is oriented for the receiver, but the receiver is
230 /// responsible for projection, etc.
231 ///
232 /// Uses:
233 /// - ConstGlobalCache: Receiver
234 /// - DataBox: system::variables_tag, Tags::Element<volume_dim>,
235 /// Tags::Extents<volume_dim>, Tags::TimeId
236 ///
237 /// DataBox changes:
238 /// - Adds: Tags::HistoryBoundaryVariables<Direction<volume_dim>,
239 /// system::variables_tag>
240 /// - Removes: nothing
241 /// - Modifies: nothing
242 ///
243 /// \tparam Receiver the parallel_component to send data to
244 template <typename Receiver>
245 struct SendDataForFluxes {
246  template <typename DbTags, typename... InboxTags, typename Metavariables,
247  typename ArrayIndex, typename ActionList>
248  static auto apply(db::DataBox<DbTags>& box,
249  tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
250  Parallel::ConstGlobalCache<Metavariables>& cache,
251  const ArrayIndex& /*array_index*/,
252  const ActionList /*meta*/) noexcept {
253  using system = typename Metavariables::system;
254  constexpr const size_t volume_dim = system::volume_dim;
255  using variables_tag = typename system::variables_tag;
256 
257  auto& receiver_proxy = Parallel::get_parallel_component<Receiver>(cache);
258 
259  const auto& element = db::get<Tags::Element<volume_dim>>(box);
260  const auto& extents = db::get<Tags::Extents<volume_dim>>(box);
261  const auto& time_id = db::get<Tags::TimeId>(box);
262  const auto& evolved_vars = db::get<variables_tag>(box);
263 
264  std::unordered_map<Direction<volume_dim>, db::item_type<variables_tag>>
265  mortars;
266 
267  for (const auto& direction_neighbors : element.neighbors()) {
268  const auto& direction = direction_neighbors.first;
269  const auto& neighbors_in_direction = direction_neighbors.second;
270 
271  const size_t dimension = direction.dimension();
272  const auto& segment_id = gsl::at(element.id().segment_ids(), dimension);
273  const auto& orientation = neighbors_in_direction.orientation();
274  const auto boundary_extents = extents.slice_away(dimension);
275  auto boundary_variables = data_on_slice(
276  evolved_vars, extents, dimension,
277  direction.side() == Side::Lower ? 0 : extents[dimension] - 1);
278 
279  for (const auto& neighbor : neighbors_in_direction) {
280  const auto& neighbor_segment_id =
281  gsl::at(neighbor.segment_ids(), dimension);
282 
283  const bool neighbor_is_on_opposite_side_of_mortar =
284  element.id().block_id() != neighbor.block_id() or
285  not segment_id.overlaps(neighbor_segment_id);
286  const auto direction_from_neighbor =
287  neighbor_is_on_opposite_side_of_mortar
288  ? orientation.mapped(direction.opposite())
289  : direction;
290  auto neighbor_variables =
291  neighbor_is_on_opposite_side_of_mortar
292  ? orient_variables_on_slice(boundary_variables,
293  boundary_extents, dimension,
294  orientation)
295  : boundary_variables;
296 
297  receiver_proxy[ElementIndex<volume_dim>(neighbor)]
298  .template receive_data<
299  typename ComputeBoundaryFlux<system>::FluxesTag>(
300  time_id,
301  std::make_pair(
302  std::make_pair(direction_from_neighbor, element.id()),
303  std::move(neighbor_variables)));
304  }
305 
306  mortars.emplace(direction, std::move(boundary_variables));
307  }
308  return std::make_tuple(
309  db::create_from<
310  db::RemoveTags<>,
311  db::AddTags<Tags::HistoryBoundaryVariables<Direction<volume_dim>,
312  variables_tag>>>(
313  box, std::move(mortars)));
314  }
315 };
316 } // namespace Actions
