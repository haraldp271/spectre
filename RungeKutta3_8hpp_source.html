107 template <typename BoundaryVars, typename FluxVars, typename Coupling>
108 BoundaryVars RungeKutta3::compute_boundary_delta(
109  const Coupling& coupling,
110  const std::vector<std::deque<std::tuple<Time, BoundaryVars, FluxVars>>>&
111  history,
112  const TimeDelta& time_step) const noexcept {
113  std::vector<std::reference_wrapper<const FluxVars>> coupling_args;
114  std::transform(history.begin(), history.end(),
115  std::inserter(coupling_args, coupling_args.begin()),
116  [](const auto& side) {
117  return std::cref(std::get<2>(side.back()));
118  });
119 
120  auto u = make_with_value<BoundaryVars>(std::get<1>(history[0][0]), 0.);
121  step_work(make_not_null(&u), history[0], time_step, coupling(coupling_args));
122  return u;
123 }

125 template <typename Vars, typename UnusedData, typename DerivVars>
126 void RungeKutta3::step_work(
127  const gsl::not_null<Vars*> u,
128  const std::deque<std::tuple<Time, Vars, UnusedData>>& history,
129  const TimeDelta& time_step,
130  const DerivVars& dt_vars) const noexcept {
131  const size_t substep = history.size() - 1;
132  const auto& vars = std::get<1>(history.back());
133  const auto& U0 = std::get<1>(history[0]);
134 
135  switch (substep) {
136  case 0: {
137  // from (5.32) of Hesthaven
138  // v^(1) = u^n + dt*RHS(u^n,t^n)
139  // On entry V = u^n, U0 = u^n, rhs0 = RHS(u^n,t^n),
140  // time = t^n
141  *u += time_step.value() * dt_vars;
142  // On exit v = v^(1), time = t^n + dt
143  return;
144  }
145  case 1: {
146  // from (5.32) of Hesthaven
147  // v^(2) = (1/4)*( 3*u^n + v^(1) + dt*RHS(v^(1),t^n + dt) )
148  // On entry V = v^(1), U0 = u^n, rhs0 = RHS(v^(1),t^n + dt),
149  // time = t^n + dt
150  *u += 0.25 * (3.0 * (U0 - vars) + time_step.value() * dt_vars);
151  // On exit v = v^(2), time = t^n + (1/2)*dt
152  return;
153  }
154  case 2: {
155  // from (5.32) of Hesthaven
156  // u^(n+1) = (1/3)*( u^n + 2*v^(2) + 2*dt*RHS(v^(2),t^n + (1/2)*dt) )
157  // On entry V = v^(2), U0 = u^n, rhs0 = RHS(v^(2),t^n + (1/2)*dt),
158  // time = t^n + (1/2)*dt
159  *u += (1.0 / 3.0) * (U0 - vars + 2.0 * time_step.value() * dt_vars);
160  // On exit v = u^(n+1), time = t^n + dt
161  return;
162  }
163  default:
164  ERROR("Bad substep value in RK3: " << substep);
165  }
166 }

177 template <typename BoundaryVars, typename FluxVars>
178 typename std::vector<typename std::deque<
179  std::tuple<Time, BoundaryVars, FluxVars>>::const_iterator>
180 RungeKutta3::needed_boundary_history(
181  const std::vector<
182  std::deque<std::tuple<Time, BoundaryVars, FluxVars>>>& history,
183  const TimeDelta& /*time_step*/) const noexcept {
184  const bool at_step_end = history[0].size() == number_of_substeps();
185  std::vector<typename std::deque<
186  std::tuple<Time, BoundaryVars, FluxVars>>::const_iterator> result;
187  result.reserve(history.size());
188  for (const auto& side_hist : history) {
189  ASSERT((side_hist.size() == number_of_substeps()) == at_step_end,
190  "Side histories inconsistent");
191  result.push_back(at_step_end ? side_hist.end() : side_hist.begin());
192  }
193  return result;
194 }
