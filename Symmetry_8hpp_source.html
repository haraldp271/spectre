16 /// \brief Metafunction that computes the new sign map
17 ///
18 /// \details
19 /// Metafunction that computes the new sign map depending on whether or not the
20 /// `index` is already in the sign map. If the current index is _not_ in the
21 /// sign map then we just return the `sign_map`. If the `index` is in the sign
22 /// map then we remove it and re-add it but mapping to -1 instead of +1.
23 /// This is how we handle anti-symmetries where it is the first index in the
24 /// user list that has the negative sign rather than the second.
25 ///
26 /// \see compute_sign_map compute_sign
27 template <bool>
28 struct compute_sign_map_helper {
29  /// \tparam SignMap the sign map to update
30  /// \tparam Index the index whose sign to replace
31  template <typename SignMap, std::int32_t Index>
32  using type = SignMap;
33 };
34 
35 template <>
36 struct compute_sign_map_helper<true> {
37  /// \tparam SignMap the sign map to update
38  /// \tparam Index the index whose sign to replace
39  template <typename SignMap, std::int32_t Index>
40  using type = tmpl::insert<
41  tmpl::erase<SignMap,
42  tmpl::abs<tmpl::int32_t<static_cast<std::int32_t>(Index)>>>,
43  tmpl::pair<tmpl::abs<tmpl::int32_t<Index>>, tmpl::int32_t<-1>>>;
44 };

48 /// \ingroup Tensor
49 /// Metafunction to compute the new `sign_map` which is a map between indices
50 /// and +/-1 depending on whether the index is symmetric (or has no symmetry) or
51 /// is anti-symmetric, respectively.
52 ///
53 /// Which metafunction implementation is called is determined by whether
54 /// `index` is already in the `sign_map` or not.
55 ///
56 /// If `index` is not in `sign_map` then we return a new `sign_map` with the
57 /// pair <index, sign<index>> added.
58 ///
59 /// In the case that `index` is in the `sign_map` we use
60 /// `compute_sign_map_helper` with the last parameter being true if the value
61 /// already in the `sign_map` is 1 AND the current index is less than 0,
62 /// otherwise it is false.
63 ///
64 /// the `bool` is
65 /// `tmpl::has_key<SignMap, tmpl::abs<tmpl::int32_t<Index>>>::value`
66 /// where `SignMap` and `Index` are the template parameters on the type alias
67 template <bool>
68 struct compute_sign_map;
69 /// \cond HIDDEN_SYMBOLS
70 template <>
71 struct compute_sign_map<true> {
72  /// \tparam SignMap map between abs(Index) and +/- 1
73  /// \tparam Index the index whose sign to possibly add to SignMap
74  template <typename SignMap, std::int32_t Index>
75  using type = typename compute_sign_map_helper<
76  tmpl::greater<tmpl::at<SignMap, tmpl::abs<tmpl::int32_t<Index>>>,
77  tmpl::int32_t<0>>::value and
78  (Index < 0)>::template type<SignMap, Index>;
79 };

93 /// \ingroup Tensor
94 /// A metafunction that, given the current and past SignMap, returns either 1
95 /// if the absolute value of index is not in the `PastSignMap` or returns the
96 /// signed value in the `CurrentSignMap` if the absolute value of index is in
97 /// the `PastSignMap` (and therefore also in the `CurrentSignMap`). The
98 /// reason why the sign from the map is returned is because it is always the
99 /// first instance of an index that has the negative sign, and since we compute
100 /// from right to left this algorithm achieves that.
101 ///
102 /// if `bool` is `true` then the `CurrentSignMap` has the value `abs<Index>` as
103 /// a key

150 /// \ingroup Tensor
151 /// Metafunction that computes symmetry as a typelist of std::integral_constant.
152 /// The `added_map` is a typemap used to keep track of whether or not a
153 /// particular index has already been added to the list. If it or its negative
154 /// (to handle anti-symmetries) has not been added then we add to the map. This
155 /// conditional add is done using tmpl::conditional_t.
156 ///
157 /// The `current_index_t` is the current symmetry index, for example value 2 in
158 /// `Symmetry<1, 2, 1>`. If the current index is already in the `next_added_map`
159 /// then we just use that value, otherwise we check if `next_added_map` has the
160 /// negative of the `index`, then use that. Finally, if `index` is not in the
161 /// `next_added_map` we have a new index given by `current_index_t` in the
162 /// `next_symm` plus one.
163 ///
164 /// The `sign_map` is used to keep track of the positive and negative signs
165 /// needed for anti-symmetries. A helper metafunction, `compute_sign_map` is
166 /// used to compute the changes to the `sign_map`. The `sign_map` is a map
167 /// between the symmetry index that is in the typelist and either plus or minus
168 /// one, depending on whether the index is symmetric or anti-symmetric
169 /// respectively.
170 ///
171 /// \tparam index the current index to add to the typelist
172 /// \tparam T the remaining indices to add to the typelist
173 template <std::int32_t index, std::int32_t... T>
174 struct symm<index, T...> {
175  static_assert(index > 0, "Anti-symmetries are currently not supported");
176  using next_symm = symm<T...>;
177  using next_added_map = typename next_symm::added_map;
178  using next_sign_map = typename next_symm::sign_map;
179 
180  using added_map = tmpl::conditional_t<
181  (tmpl::has_key<next_added_map, tmpl::int32_t<index>>::value or
182  tmpl::has_key<next_added_map, tmpl::int32_t<-index>>::value),
183  next_added_map,
184  tmpl::insert<
185  next_added_map,
186  tmpl::pair<tmpl::int32_t<index>,
187  tmpl::int32_t<next_symm::current_index_t::value + 1>>>>;
188 
189  using current_index_t = tmpl::conditional_t<
190  tmpl::has_key<next_added_map, tmpl::int32_t<index>>::value,
191  tmpl::at<next_added_map, tmpl::int32_t<index>>,
192  tmpl::conditional_t<
193  tmpl::has_key<next_added_map, tmpl::int32_t<-index>>::value,
194  tmpl::at<next_added_map, tmpl::int32_t<-index>>,
195  tmpl::int32_t<next_symm::current_index_t::value + 1>>>;
196 
197  using sign_map = typename compute_sign_map<
198  tmpl::has_key<next_sign_map, tmpl::abs<tmpl::int32_t<index>>>::value>::
199  template type<next_sign_map, index>;
200 
201  using type = tmpl::push_front<
202  typename next_symm::type,
203  tmpl::int32_t<
204  compute_sign<
205  tmpl::has_key<sign_map, tmpl::abs<tmpl::int32_t<index>>>::value>::
206  template type<sign_map, next_sign_map, index>::type::value *
207  current_index_t::value>>;
208 };

211 /// \ingroup Tensor
212 /// \brief Computes the canonical symmetry from the integers `T`
213 ///
214 /// \details
215 /// Compute the canonical symmetry typelist given a set of integers, T. The
216 /// resulting typelist is in descending order of the absolute value of the
217 /// integers. For example, the result of `Symmetry<1, 2, 3>` is
218 /// `integral_list<int32_t, 3, 2, 1>`. Anti-symmetries can be denoted with a
219 /// minus sign on either _or_ both indices. That is, `Symmetry<-1, 2, 1>` is
220 /// anti-symmetric in the first and last index and is the same as `Symmetry<-1,
221 /// 2, -1>`. Note: two minus signs are still anti-symmetric because it
222 /// simplifies the algorithm used to compute the canonical form of the symmetry.
223 ///
224 /// \tparam T the integers denoting the symmetry of the Tensor
225 template <std::int32_t... T>
226 using Symmetry = typename detail::symm<T...>::type;
