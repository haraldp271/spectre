13 /*!
14  * \brief Conjugate momentum to the spacetime metric.
15  *
16  * \details If \f$ \psi_{ab} \f$ is the spacetime metric, and \f$ N \f$ and
17  * \f$ N^i \f$ are the lapse and shift respectively, then we define
18  * \f$ \Pi_{ab} = -\frac{1}{N} ( \partial_t \psi_{ab} + N^{i} \phi_{iab} ) \f$
19  * where \f$\phi_{iab}\f$ is the variable defined by the tag Phi.
20  */

22 struct Pi : db::DataBoxTag {
