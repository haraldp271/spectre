85 /*!
86  * \ingroup CoordinateMapsGroup
87  * \brief A coordinate map or composition of coordinate maps
88  *
89  * Maps coordinates from the `SourceFrame` to the `TargetFrame` using the
90  * coordinate maps `Maps...`. The individual maps are applied left to right
91  * from the source to the target Frame. The inverse map, as well as Jacobian
92  * and inverse Jacobian are also provided. The `CoordinateMap` class must
93  * be used even if just wrapping a single coordinate map. It is designed to
94  * be an extremely minimal interface to the underlying coordinate maps. For
95  * a list of all coordinate maps see the CoordinateMaps group or namespace.
96  *
97  * Each coordinate map must contain a `static constexpr size_t dim` variable
98  * that is equal to the dimensionality of the map. The Coordinatemap class
99  * contains a member `static constexpr size_t dim`, a type alias `source_frame`,
100  * a type alias `target_frame` and `typelist of the `Maps...`.
101  */
102 template <typename SourceFrame, typename TargetFrame, typename... Maps>
103 class CoordinateMap
104  : public CoordinateMapBase<SourceFrame, TargetFrame,
105  CoordinateMaps::map_dim<Maps...>> {
106  static_assert(sizeof...(Maps) > 0, "Must have at least one map");
107  static_assert(
108  tmpl::all<tmpl::integral_list<size_t, Maps::dim...>,
109  std::is_same<tmpl::integral_constant<
110  size_t, CoordinateMaps::map_dim<Maps...>>,
111  tmpl::_1>>::value,
112  "All Maps passed to CoordinateMap must be of the same dimensionality.");
113 
114  public:
115  static constexpr size_t dim = CoordinateMaps::map_dim<Maps...>;
116  using source_frame = SourceFrame;

251 template <typename SourceFrame, typename TargetFrame, typename... Maps>
252 template <typename T>
253 constexpr SPECTRE_ALWAYS_INLINE tnsr::I<
254  T, CoordinateMap<SourceFrame, TargetFrame, Maps...>::dim, TargetFrame>
255 CoordinateMap<SourceFrame, TargetFrame, Maps...>::call_impl(
256  const tnsr::I<T, dim, SourceFrame>& source_point) const {
257  std::array<T, dim> mapped_point = make_array<T, dim>(source_point);
258  tuple_fold(maps_, [](const auto& map,
259  std::array<T, dim>& point) { point = map(point); },
260  mapped_point);
261  return tnsr::I<T, dim, TargetFrame>(std::move(mapped_point));
262 }

265 template <typename T>
266 constexpr SPECTRE_ALWAYS_INLINE tnsr::I<
267  T, CoordinateMap<SourceFrame, TargetFrame, Maps...>::dim, SourceFrame>
268 CoordinateMap<SourceFrame, TargetFrame, Maps...>::inverse_impl(
269  const tnsr::I<T, dim, TargetFrame>& target_point) const {
270  std::array<T, dim> mapped_point = make_array<T, dim>(target_point);
271  tuple_fold<true>(maps_,
272  [](const auto& map, std::array<T, dim>& point) {

274  },

279 template <typename SourceFrame, typename TargetFrame, typename... Maps>
280 template <typename T>
281 constexpr SPECTRE_ALWAYS_INLINE auto
282 CoordinateMap<SourceFrame, TargetFrame, Maps...>::inv_jacobian_impl(
283  const tnsr::I<T, dim, SourceFrame>& source_point) const
284  -> Tensor<T, tmpl::integral_list<std::int32_t, 2, 1>,
285  index_list<SpatialIndex<dim, UpLo::Up, SourceFrame>,
286  SpatialIndex<dim, UpLo::Lo, TargetFrame>>> {
287  std::array<T, dim> mapped_point = make_array<T, dim>(source_point);
288 
289  Tensor<T, tmpl::integral_list<std::int32_t, 2, 1>,
290  index_list<SpatialIndex<dim, UpLo::Up, SourceFrame>,
291  SpatialIndex<dim, UpLo::Lo, TargetFrame>>>
292  inv_jac{};
293 
294  tuple_transform(
295  maps_,
296  [&inv_jac, &mapped_point](const auto& map, auto index,
297  const std::tuple<Maps...>& maps) {
298  constexpr const size_t count = decltype(index)::value;
299 
300  if (LIKELY(count != 0)) {
301  mapped_point =
302  std::get<(count != 0 ? count - 1 : 0)>(maps)(mapped_point);
303  const auto temp_inv_jac = map.inv_jacobian(mapped_point);
304  std::array<T, dim> temp{};
305  for (size_t source = 0; source < dim; ++source) {
306  for (size_t target = 0; target < dim; ++target) {
307  gsl::at(temp, target) =
308  inv_jac.get(source, 0) * temp_inv_jac.get(0, target);
309  for (size_t dummy = 1; dummy < dim; ++dummy) {
310  gsl::at(temp, target) += inv_jac.get(source, dummy) *
311  temp_inv_jac.get(dummy, target);
312  }
313  }
314  for (size_t target = 0; target < dim; ++target) {
315  inv_jac.get(source, target) = std::move(gsl::at(temp, target));
316  }
317  }
318  } else {
319  auto temp_inv_jac = map.inv_jacobian(mapped_point);
320  for (size_t source = 0; source < dim; ++source) {
321  for (size_t target = 0; target < dim; ++target) {
322  inv_jac.get(source, target) =
323  std::move(temp_inv_jac.get(source, target));
324  }
325  }
326  }
327  },
328  maps_);
329  return inv_jac;
330 }

332 template <typename SourceFrame, typename TargetFrame, typename... Maps>
333 template <typename T>
334 constexpr SPECTRE_ALWAYS_INLINE auto
335 CoordinateMap<SourceFrame, TargetFrame, Maps...>::jacobian_impl(
336  const tnsr::I<T, dim, SourceFrame>& source_point) const
337  -> Tensor<T, tmpl::integral_list<std::int32_t, 2, 1>,
338  index_list<SpatialIndex<dim, UpLo::Up, TargetFrame>,
339  SpatialIndex<dim, UpLo::Lo, SourceFrame>>> {
340  std::array<T, dim> mapped_point = make_array<T, dim>(source_point);
341  Tensor<T, tmpl::integral_list<std::int32_t, 2, 1>,
342  index_list<SpatialIndex<dim, UpLo::Up, TargetFrame>,
343  SpatialIndex<dim, UpLo::Lo, SourceFrame>>>
344  jac{};
345  tuple_transform(
346  maps_,
347  [&jac, &mapped_point](const auto& map, auto index,
348  const std::tuple<Maps...>& maps) {
349  constexpr const size_t count = decltype(index)::value;
350 
351  if (LIKELY(count != 0)) {
352  mapped_point =
353  std::get<(count != 0 ? count - 1 : 0)>(maps)(mapped_point);
354  const auto noframe_jac = map.jacobian(mapped_point);
355  std::array<T, dim> temp{};
356  for (size_t source = 0; source < dim; ++source) {
357  for (size_t target = 0; target < dim; ++target) {
358  gsl::at(temp, target) =
359  noframe_jac.get(target, 0) * jac.get(0, source);
360  for (size_t dummy = 1; dummy < dim; ++dummy) {
361  gsl::at(temp, target) +=
362  noframe_jac.get(target, dummy) * jac.get(dummy, source);
363  }
364  }
365  for (size_t target = 0; target < dim; ++target) {
366  jac.get(target, source) = std::move(gsl::at(temp, target));
367  }
368  }
369  } else {
370  auto noframe_jac = map.jacobian(mapped_point);
371  for (size_t target = 0; target < dim; ++target) {
372  for (size_t source = 0; source < dim; ++source) {
373  jac.get(target, source) =
374  std::move(noframe_jac.get(target, source));
375  }
376  }
377  }
378  },
379  maps_);
380  return jac;
381 }
