20 /// \ingroup ConstantExpressions
21 /// Compute 2 to the n for integral types.
22 ///
23 /// \param n the power of two to compute.
24 /// \return 2^n
25 template <typename T,
26  Requires<tt::is_integer_v<T> and cpp17::is_unsigned_v<T>> = nullptr>
27 SPECTRE_ALWAYS_INLINE constexpr T two_to_the(T n) {
28  CASSERT(n < 8 * sizeof(T), "two_to_the is overflowing!");
29  return T(1) << n;
30 }
